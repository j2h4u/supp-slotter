    **`load_substance_registry`** (planner/cards/substance.py:275–284):
    Track total scanned and skipped counts. Print a summary to stderr after the loop iff skip_count > 0.
    ```python
    def load_substance_registry() -> dict[str, Substance]:
        substances: dict[str, Substance] = {}
        paths = sorted(SUBSTANCES_DIR.glob("*.yaml"))
        skipped = 0
        for sf in paths:
            try:
                substance = load_substance(sf)
            except CardLoadError as e:
                print(f"warning: skipping substance card: {e.message}", file=sys.stderr)
                skipped += 1
                continue
            substances[substance.id] = substance
        if skipped:
            print(
                f"warning: loaded {len(substances)}/{len(paths)} substance cards; {skipped} skipped",
                file=sys.stderr,
            )
        return substances
    ```

    **`load_product_registry`** (planner/cards/product.py:160–169):
    Same pattern, mirroring substance text:
    ```python
    def load_product_registry() -> dict[str, Product]:
        products: dict[str, Product] = {}
        paths = sorted(PRODUCTS_DIR.glob("*.yaml"))
        skipped = 0
        for pf in paths:
            try:
                product = load_product(pf)
            except CardLoadError as e:
                print(f"warning: skipping product card: {e.message}", file=sys.stderr)
                skipped += 1
                continue
            products[product.id] = product
        if skipped:
            print(
                f"warning: loaded {len(products)}/{len(paths)} product cards; {skipped} skipped",
                file=sys.stderr,
            )
        return products
    ```
